void loadData()
{
	FILE* fp = fopen("D:/demo2/atm.txt", "r");
	if (fp != NULL)
	{
		while (!feof(fp))
		{
			//创建结点
			Account* newNode = (Account*)malloc(sizeof(Account));
			if (newNode == NULL) 
			{
				exit(-1);
			}//判断malloc申请空间是否成功，消除警告

			//结点初始化
			newNode->next = NULL;
			fscanf_s(fp, "%s %s %f\n", newNode->username,19, newNode->password, 7, &newNode->money);

			//添加结点到链表
			if (head == NULL)
			{
				head = newNode;
				tail = newNode;
			}
			else
			{
				tail->next = newNode;
				tail = newNode;
			}
		}
		fclose(fp);
	}
}
